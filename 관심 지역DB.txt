drop table FAVORITE_STATION;

CREATE SEQUENCE SEQ_FAVORITE_STATION START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

ALTER TABLE FAVORITE_STATION MODIFY station_id NULL;
ALTER TABLE FAVORITE_STATION DROP CONSTRAINT fk_favorite_station;

CREATE TABLE FAVORITE_STATION (
  favorite_id     NUMBER PRIMARY KEY,
  user_id         VARCHAR2(64)  NOT NULL,
  station_id      NUMBER NOT NULL,
  station_name    VARCHAR2(200),
  created_at      TIMESTAMP DEFAULT SYSTIMESTAMP NOT NULL,
  CONSTRAINT fk_favorite_station FOREIGN KEY (station_id)
      REFERENCES STATION_INFO(station_id)
);
commit;
